query3 = '''
    SELECT days_interval, AVG(likes) AS likes
    FROM 
        (SELECT 
            julianday(date) - julianday((SELECT MAX(date) FROM df WHERE date < t.date)) AS days_interval,
            likes
        FROM 
            df t
        ) t
    GROUP BY days_interval;
'''
result3 = sqldf(query3, locals())
