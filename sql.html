query = '''

SELECT b.book_id,
       b.title,
       COUNT(DISTINCT re.review_id) AS rev,
       ROUND(AVG(r.rating),2) AS rat
       
FROM books AS b
LEFT JOIN ratings AS r ON b.book_id=r.book_id
LEFT JOIN reviews AS re ON b.book_id=re.book_id

GROUP BY b.book_id,
         b.title         
ORDER BY rat DESC;

query = '''
WITH tab AS

(SELECT book_id,
       publisher_id
FROM books 
WHERE num_pages > 50)

SELECT tab.publisher_id,
       pb.publisher
FROM tab
JOIN publishers AS pb ON tab.publisher_id=pb.publisher_id

GROUP BY tab.publisher_id, pb.publisher

query = '''

SELECT COUNT(DISTINCT rv.text)/ COUNT(DISTINCT rv.username) AS avg_rv_count
FROM ratings AS r

INNER JOIN reviews AS rv
ON r.username = rv.username

WHERE r.username in ( 
                    SELECT username
                    FROM ratings
                    GROUP BY username
                    HAVING COUNT(rating_id) > 50
                    )
